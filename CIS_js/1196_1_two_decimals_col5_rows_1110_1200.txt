(function () {
    var lookupByStaticInputId = {};
    var allowed1000 = {};
    var allowed1300 = {};

    CAP1196_ROWIDS_1000.forEach(function (row) {
        lookupByStaticInputId["73_1196_" + row.id + "_" + row.rind + "_0"] = row.rind;
        allowed1000[row.rind] = true;
    });

    CAP1196_ROWIDS_1300.forEach(function (row) {
        lookupByStaticInputId["73_1196_" + row.id + "_" + row.rind + "_0"] = row.rind;
        allowed1300[row.rind] = true;
    });

    function normalizeRind(value) {
        return $.trim(String(value || "")).replace(/\s+/g, "");
    }

    function getValidationInputs() {
        return $("input[id^='73_1196_'][id$='_0'], input[id^='T1_']").toArray();
    }

    function getOrCreateInlineContainer(input) {
        var id = input.id + "_native_inline_error";
        var node = document.getElementById(id);

        if (!node) {
            node = document.createElement("div");
            node.id = id;
            node.style.color = "#d93025";
            node.style.fontSize = "11px";
            node.style.lineHeight = "1.2";
            node.style.marginTop = "2px";
            node.style.whiteSpace = "normal";
            node.style.maxWidth = "180px";
            input.parentNode.appendChild(node);
        }

        return node;
    }

    function renderMessages(input, messages) {
        var container = document.getElementById(input.id + "_native_inline_error");

        if (!messages || !messages.length) {
            if (container && container.parentNode) {
                container.parentNode.removeChild(container);
            }
            input.style.border = "";
            input.style.backgroundColor = "";
            return;
        }

        container = getOrCreateInlineContainer(input);
        container.innerHTML = "";

        messages.forEach(function (message) {
            var row = document.createElement("div");
            row.textContent = message;
            container.appendChild(row);
        });

        input.style.border = "2px solid #d93025";
        input.style.backgroundColor = "#fff5f5";
    }

    function getDynamicGroup(input) {
        var $row = $(input).closest("tr");
        if (!$row.length) {
            return null;
        }

        var rowIndex = $row[0].rowIndex;
        var row1000 = document.getElementById("113835");
        var row1100 = document.getElementById("113830");
        var row1300 = document.getElementById("113768");
        var row1400 = document.getElementById("113898");

        if (row1000 && row1100 && rowIndex > row1000.rowIndex && rowIndex < row1100.rowIndex) {
            return "1000";
        }

        if (row1300 && row1400 && rowIndex > row1300.rowIndex && rowIndex < row1400.rowIndex) {
            return "1300";
        }

        return null;
    }

    function buildDynamicMessage(group, entered) {
        if (group === "1000") {
            return "Cod rând incorect. În acest bloc sunt permise doar coduri din seria 1000-1 ... 1000-60. Ai introdus: " + entered + ".";
        }

        if (group === "1300") {
            return "Cod rând incorect. În acest bloc sunt permise doar coduri din seria 1300-1 ... 1300-62. Ai introdus: " + entered + ".";
        }

        return "Cod rând incorect. Ai introdus: " + entered + ".";
    }

    function buildDuplicateMessage(entered) {
        return "Codul rândului se repetă. Ai introdus de mai multe ori: " + entered + ".";
    }

    function revalidateAllCap1196Inputs() {
        var inputs = getValidationInputs();
        var state = {};
        var duplicates = {};

        inputs.forEach(function (input) {
            var entered = normalizeRind(input.value);
            var expectedStatic = lookupByStaticInputId[input.id] || null;
            var group = /^T1_\d+$/.test(input.id) ? getDynamicGroup(input) : null;

            state[input.id] = {
                input: input,
                entered: entered,
                expectedStatic: expectedStatic,
                group: group,
                messages: []
            };

            if (entered !== "") {
                if (!duplicates[entered]) {
                    duplicates[entered] = [];
                }
                duplicates[entered].push(input.id);
            }
        });

        Object.keys(state).forEach(function (inputId) {
            var item = state[inputId];

            if (item.entered === "") {
                return;
            }

            if (item.expectedStatic) {
                if (item.entered !== normalizeRind(item.expectedStatic)) {
                    item.messages.push(
                        "Cod rând incorect. Trebuie exact: " + item.expectedStatic + ". Ai introdus: " + item.entered + "."
                    );
                }
            } else if (/^T1_\d+$/.test(item.input.id)) {
                if (item.group === "1000" && !allowed1000[item.entered]) {
                    item.messages.push(buildDynamicMessage(item.group, item.entered));
                }

                if (item.group === "1300" && !allowed1300[item.entered]) {
                    item.messages.push(buildDynamicMessage(item.group, item.entered));
                }
            }
        });

        Object.keys(duplicates).forEach(function (code) {
            if (duplicates[code].length > 1) {
                duplicates[code].forEach(function (inputId) {
                    state[inputId].messages.push(buildDuplicateMessage(code));
                });
            }
        });

        Object.keys(state).forEach(function (inputId) {
            renderMessages(state[inputId].input, state[inputId].messages);
        });
    }

    window.revalidateAllCap1196Inputs = revalidateAllCap1196Inputs;

    $(document).on(
        "input keyup change",
        "input[id^='73_1196_'][id$='_0'], input[id^='T1_']",
        function () {
            revalidateAllCap1196Inputs();
        }
    );

    $(document).on(
        "blur",
        "input[id^='73_1196_'][id$='_0'], input[id^='T1_']",
        function () {
            setTimeout(function () {
                revalidateAllCap1196Inputs();
            }, 30);
        }
    );
})();

(function () {
    var cap1196TwoDecimalTargets = {
        "73_1196_113831_1110_5": true,
        "73_1196_113832_1120_5": true,
        "73_1196_113833_1130_5": true,
        "73_1196_113769_1140_5": true,
        "73_1196_113717_1150_5": true,
        "73_1196_113641_1160_5": true,
        "73_1196_113620_1170_5": true,
        "73_1196_113684_1180_5": true,
        "73_1196_113685_1190_5": true,
        "73_1196_113834_1200_5": true
    };

    function isCap1196TwoDecimalTarget(el) {
        return !!(el && el.id && cap1196TwoDecimalTargets[el.id]);
    }

    function isAllowedTwoDecimalControlKey(e) {
        var key = e.key;
        return (
            key === "Backspace" ||
            key === "Delete" ||
            key === "Tab" ||
            key === "Escape" ||
            key === "Enter" ||
            key === "ArrowLeft" ||
            key === "ArrowRight" ||
            key === "ArrowUp" ||
            key === "ArrowDown" ||
            key === "Home" ||
            key === "End"
        );
    }

    function sanitizeTwoDecimalValue(value) {
        value = String(value || "").replace(/,/g, ".");
        value = value.replace(/[^0-9.]/g, "");

        var parts = value.split(".");
        var intPart = parts.shift() || "";
        var decPart = parts.join("").slice(0, 2);

        if (intPart === "" && decPart === "") {
            return "";
        }

        return decPart.length ? intPart + "." + decPart : intPart;
    }

    $(document).on("keydown", "input", function (e) {
        if (!isCap1196TwoDecimalTarget(this)) {
            return;
        }

        if (e.ctrlKey || e.metaKey || e.altKey) {
            return;
        }

        if (isAllowedTwoDecimalControlKey(e)) {
            return;
        }

        if (/^[0-9]$/.test(e.key)) {
            return;
        }

        if ((e.key === "." || e.key === ",") && this.value.indexOf(".") === -1 && this.value.indexOf(",") === -1) {
            return;
        }

        e.preventDefault();
    });

    $(document).on("input blur paste", "input", function () {
        if (!isCap1196TwoDecimalTarget(this)) {
            return;
        }

        var cleaned = sanitizeTwoDecimalValue(this.value);
        if (this.value !== cleaned) {
            this.value = cleaned;
        }
    });
})();

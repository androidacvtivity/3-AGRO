// PATCH robust pentru afisarea imediata a erorii la codul randului
// Se lipeste DUPA declararea listelor CAP1196_ROWIDS_1000 / CAP1196_ROWIDS_1300
// si dupa functiile de autosuma.

(function () {
    var allRows = [];
    if (typeof CAP1196_ROWIDS_1000 !== 'undefined' && Array.isArray(CAP1196_ROWIDS_1000)) {
        allRows = allRows.concat(CAP1196_ROWIDS_1000);
    }
    if (typeof CAP1196_ROWIDS_1300 !== 'undefined' && Array.isArray(CAP1196_ROWIDS_1300)) {
        allRows = allRows.concat(CAP1196_ROWIDS_1300);
    }

    var lookupByInputId = {};
    allRows.forEach(function (row) {
        lookupByInputId['73_1196_' + row.id + '_' + row.rind + '_0'] = row.rind;
    });

    function normalizeRind(value) {
        return $.trim(String(value || '')).replace(/\s+/g, '');
    }

    function getErrorMessage(expected, actual) {
        return 'Cod rând incorect. Trebuie: ' + expected + '. Ai introdus: ' + actual + '.';
    }

    function getOrCreateInlineMessage(input) {
        var id = input.id + '_native_inline_error';
        var node = document.getElementById(id);

        if (!node) {
            node = document.createElement('div');
            node.id = id;
            node.style.color = '#d93025';
            node.style.fontSize = '11px';
            node.style.lineHeight = '1.2';
            node.style.marginTop = '2px';
            node.style.whiteSpace = 'normal';
            node.style.maxWidth = '140px';
            input.parentNode.appendChild(node);
        }

        return node;
    }

    function clearInlineMessage(input) {
        var node = document.getElementById(input.id + '_native_inline_error');
        if (node) {
            node.parentNode.removeChild(node);
        }
    }

    function showTopMessage(message) {
        var top = document.getElementById('InputAlertMsg');
        if (top) {
            top.innerHTML = '<span style="color:#d93025;font-weight:bold;">' + message + '</span>';
        }
    }

    function clearTopMessage() {
        var top = document.getElementById('InputAlertMsg');
        if (top) {
            top.innerHTML = '';
        }
    }

    function showGrowl(message) {
        if (typeof $.bootstrapGrowl === 'function') {
            $.bootstrapGrowl(message, {
                type: 'danger',
                delay: 2500,
                allow_dismiss: true,
                placement: { from: 'top', align: 'right' },
                offset: { from: 'top', amount: 60 }
            });
        }
    }

    function markInvalid(input, message, showPopup) {
        input.style.border = '2px solid #d93025';
        input.style.backgroundColor = '#fff5f5';

        var inlineNode = getOrCreateInlineMessage(input);
        inlineNode.textContent = message;

        showTopMessage(message);

        if (typeof input.setCustomValidity === 'function') {
            input.setCustomValidity(message);
            if (showPopup && typeof input.reportValidity === 'function') {
                input.reportValidity();
            }
        }
    }

    function clearInvalid(input) {
        input.style.border = '';
        input.style.backgroundColor = '';
        clearInlineMessage(input);

        if (typeof input.setCustomValidity === 'function') {
            input.setCustomValidity('');
        }

        clearTopMessage();
    }

    function validateCap1196Input(input, showPopup) {
        if (!input || !input.id) {
            return true;
        }

        var expected = lookupByInputId[input.id];
        if (!expected) {
            return true;
        }

        var entered = normalizeRind(input.value);
        var expectedNorm = normalizeRind(expected);

        if (entered !== '' && entered !== expectedNorm) {
            var message = getErrorMessage(expected, entered);
            markInvalid(input, message, showPopup === true);
            return false;
        }

        clearInvalid(input);
        return true;
    }

    $(document).on('input keyup change', "input[id^='73_1196_'][id$='_0']", function () {
        validateCap1196Input(this, false);
    });

    $(document).on('blur', "input[id^='73_1196_'][id$='_0']", function () {
        var self = this;
        setTimeout(function () {
            var ok = validateCap1196Input(self, true);
            if (!ok) {
                showGrowl(self.validationMessage || 'Cod rând incorect');
            }
        }, 30);
    });
})();

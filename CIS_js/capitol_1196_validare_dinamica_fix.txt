(function () {
    var lookupByStaticInputId = {};
    var allowed1000 = {};
    var allowed1300 = {};

    CAP1196_ROWIDS_1000.forEach(function (row) {
        lookupByStaticInputId["73_1196_" + row.id + "_" + row.rind + "_0"] = row.rind;
        allowed1000[row.rind] = true;
    });

    CAP1196_ROWIDS_1300.forEach(function (row) {
        lookupByStaticInputId["73_1196_" + row.id + "_" + row.rind + "_0"] = row.rind;
        allowed1300[row.rind] = true;
    });

    function normalizeRind(value) {
        return $.trim(String(value || "")).replace(/\s+/g, "");
    }

    function getOrCreateInlineMessage(input) {
        var id = input.id + "_native_inline_error";
        var node = document.getElementById(id);

        if (!node) {
            node = document.createElement("div");
            node.id = id;
            node.style.color = "#d93025";
            node.style.fontSize = "11px";
            node.style.lineHeight = "1.2";
            node.style.marginTop = "2px";
            node.style.whiteSpace = "normal";
            node.style.maxWidth = "160px";
            input.parentNode.appendChild(node);
        }

        return node;
    }

    function clearInlineMessage(input) {
        var node = document.getElementById(input.id + "_native_inline_error");
        if (node && node.parentNode) {
            node.parentNode.removeChild(node);
        }
    }

    function showTopMessage(message) {
        var top = document.getElementById("InputAlertMsg");
        if (top) {
            top.innerHTML = '<span style="color:#d93025;font-weight:bold;">' + message + '</span>';
        }
    }

    function clearTopMessage() {
        var top = document.getElementById("InputAlertMsg");
        if (top) {
            top.innerHTML = "";
        }
    }

    function showGrowl(message) {
        if (typeof $.bootstrapGrowl === "function") {
            $.bootstrapGrowl(message, {
                type: "danger",
                delay: 3000,
                allow_dismiss: true,
                placement: { from: "top", align: "right" },
                offset: { from: "top", amount: 60 }
            });
        }
    }

    function markInvalid(input, message, showPopup) {
        input.style.border = "2px solid #d93025";
        input.style.backgroundColor = "#fff5f5";

        var inlineNode = getOrCreateInlineMessage(input);
        inlineNode.textContent = message;

        showTopMessage(message);

        if (typeof input.setCustomValidity === "function") {
            input.setCustomValidity(message);
            if (showPopup && typeof input.reportValidity === "function") {
                input.reportValidity();
            }
        }
    }

    function clearInvalid(input) {
        input.style.border = "";
        input.style.backgroundColor = "";
        clearInlineMessage(input);
        if (typeof input.setCustomValidity === "function") {
            input.setCustomValidity("");
        }
        clearTopMessage();
    }

    function getDynamicGroup(input) {
        var $row = $(input).closest("tr");
        if (!$row.length) {
            return null;
        }

        var rowIndex = $row[0].rowIndex;
        var row1000 = document.getElementById("113835");
        var row1100 = document.getElementById("113830");
        var row1300 = document.getElementById("113768");
        var row1400 = document.getElementById("113898");

        if (row1000 && row1100 && rowIndex > row1000.rowIndex && rowIndex < row1100.rowIndex) {
            return "1000";
        }

        if (row1300 && row1400 && rowIndex > row1300.rowIndex && rowIndex < row1400.rowIndex) {
            return "1300";
        }

        return null;
    }

    function buildDynamicMessage(group, entered) {
        if (group === "1000") {
            return "Cod rând incorect. Acest rând dinamic este creat în blocul 1000, de aceea aici sunt permise doar coduri din seria 1000-1 ... 1000-60. Ai introdus: " + entered + ".";
        }

        if (group === "1300") {
            return "Cod rând incorect. Acest rând dinamic este creat în blocul 1300, de aceea aici sunt permise doar coduri din seria 1300-1 ... 1300-62. Ai introdus: " + entered + ".";
        }

        return "Cod rând incorect. Ai introdus: " + entered + ".";
    }

    function validateCap1196Input(input, showPopup) {
        if (!input || !input.id) {
            return true;
        }

        var entered = normalizeRind(input.value);
        if (entered === "") {
            clearInvalid(input);
            return true;
        }

        var expectedStatic = lookupByStaticInputId[input.id];
        if (expectedStatic) {
            if (entered !== normalizeRind(expectedStatic)) {
                markInvalid(input, "Cod rând incorect. Trebuie exact: " + expectedStatic + ". Ai introdus: " + entered + ".", showPopup === true);
                return false;
            }

            clearInvalid(input);
            return true;
        }

        if (/^T1_\d+$/.test(input.id)) {
            var group = getDynamicGroup(input);

            if (group === "1000" && !allowed1000[entered]) {
                markInvalid(input, buildDynamicMessage(group, entered), showPopup === true);
                return false;
            }

            if (group === "1300" && !allowed1300[entered]) {
                markInvalid(input, buildDynamicMessage(group, entered), showPopup === true);
                return false;
            }
        }

        clearInvalid(input);
        return true;
    }

    $(document).on("input keyup change", "input[id^='73_1196_'][id$='_0'], input[id^='T1_']", function () {
        validateCap1196Input(this, false);
    });

    $(document).on("blur", "input[id^='73_1196_'][id$='_0'], input[id^='T1_']", function () {
        var self = this;
        setTimeout(function () {
            var ok = validateCap1196Input(self, true);
            if (!ok) {
                showGrowl(self.validationMessage || "Cod rând incorect");
            }
        }, 30);
    });
})();

function f_Capitol_1196() {
    sumCap1196_Rind1000();
    sumCap1196_Rind1300();
    sumCap1196_Rind1100();
    sumCap1196_Rind1900();
}

        "input:not([type='button']):not([readonly]):not([disabled])",
